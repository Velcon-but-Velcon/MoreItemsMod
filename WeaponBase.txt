using Terraria.ID;
using Terraria.ModLoader;
using Terraria;
namespace CodeBase.Items
{
	public class WeaponBase : ModItem
	{
		public override void SetStaticDefaults()
		{
			DisplayName.SetDefault("Base Melee Weapon");
			Tooltip.SetDefault("'hmm, yes I guess you've learned something.'");
		}

		public override void SetDefaults()
		{
			item.damage = 36;
			item.melee = true;
			item.width = 20;
			item.height = 20;
			item.scale = 4;
			item.useTime = 25;
			item.useAnimation = 20;
			item.useStyle = 1;
			item.knockBack = 4;
			item.value = 10000;
			item.rare = ItemRarityID.Blue;
			item.UseSound = SoundID.Item1;
			item.autoReuse = true;
			item.useTurn = true;
			item.shoot = ProjectileID.SwordBeam;
			item.shootSpeed = 10f;
		}

		public override void AddRecipes()
		{
			ModRecipe recipe = new ModRecipe(mod);
			recipe.AddIngredient(ItemID.Wood, 50);
			recipe.AddTile(TileID.DemonAltar);
			recipe.SetResult(this);
			recipe.AddRecipe();
		}
	}
}
